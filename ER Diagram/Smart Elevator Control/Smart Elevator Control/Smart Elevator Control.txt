// =========================
// TABLES
// =========================

buildings [icon:building, color: blue] {
  building_id serial pk
  name varchar(150)
  location varchar(255)
  building_type enum('COMMERCIAL','MALL','HOSPITAL','RESIDENTIAL','AIRPORT')

  created_at timestamp
}

floors [icon:layers, color: teal] {
  floor_id serial pk
  building_id int fk
  floor_number int
  label varchar(10)

  created_at timestamp
}

shafts [icon:box, color: purple] {
  shaft_id serial pk
  building_id int fk
  shaft_code varchar(50)
}

elevators [icon:move-vertical, color: green] {
  elevator_id serial pk
  building_id int fk
  shaft_id int fk
  elevator_code varchar(50)
  capacity int
  installation_date date
  is_active boolean

  created_at timestamp
}

elevator_floor_map [icon:shuffle, color: orange] {
  id serial pk
  elevator_id int fk
  floor_id int fk
}

floor_requests [icon:arrow-up-circle, color: red] {
  request_id serial pk
  building_id int fk
  source_floor_id int fk
  destination_floor_id int fk
  request_time timestamp
  direction enum('UP','DOWN')
  status enum('PENDING','ASSIGNED','COMPLETED','CANCELLED')
}

ride_assignments [icon:link, color: yellow] {
  assignment_id serial pk
  request_id int fk unique
  elevator_id int fk
  assigned_time timestamp
}

ride_logs [icon:activity, color: pink] {
  ride_id serial pk
  elevator_id int fk
  request_id int fk
  start_floor_id int fk
  end_floor_id int fk
  start_time timestamp
  end_time timestamp
  duration int
}

elevator_status_logs [icon:cpu, color: brown] {
  status_id serial pk
  elevator_id int fk
  status enum('IDLE','MOVING','MAINTENANCE','OFFLINE')
  current_floor_id int fk
  recorded_at timestamp
}

maintenance_logs [icon:tool, color: gray] {
  maintenance_id serial pk
  elevator_id int fk
  issue_description text
  maintenance_type enum('SCHEDULED','BREAKDOWN')
  start_time timestamp
  end_time timestamp
  status enum('OPEN','IN_PROGRESS','COMPLETED')
  technician_name varchar(150)
}


// =========================
// RELATIONSHIPS
// =========================

buildings.building_id < floors.building_id
buildings.building_id < shafts.building_id
buildings.building_id < elevators.building_id

shafts.shaft_id < elevators.shaft_id

elevators.elevator_id < elevator_floor_map.elevator_id
floors.floor_id < elevator_floor_map.floor_id

buildings.building_id < floor_requests.building_id
floors.floor_id < floor_requests.source_floor_id
floors.floor_id < floor_requests.destination_floor_id

floor_requests.request_id - ride_assignments.request_id
elevators.elevator_id < ride_assignments.elevator_id

elevators.elevator_id < ride_logs.elevator_id
floor_requests.request_id < ride_logs.request_id
floors.floor_id < ride_logs.start_floor_id
floors.floor_id < ride_logs.end_floor_id

elevators.elevator_id < elevator_status_logs.elevator_id
floors.floor_id < elevator_status_logs.current_floor_id

elevators.elevator_id < maintenance_logs.elevator_id
