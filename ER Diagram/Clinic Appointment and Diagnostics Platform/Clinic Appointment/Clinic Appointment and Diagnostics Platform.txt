patients [icon:user, color: blue] {
  patient_id serial pk
  name varchar(100)
  dob date
  gender varchar(10)
  phone varchar(20)
  address text

  created_at timestamp
}

specializations [icon:activity, color: purple] {
  specialization_id serial pk
  name varchar(100)
}

doctors [icon:user-check, color: green] {
  doctor_id serial pk
  name varchar(100)
  email varchar(150) unique
  phone varchar(20)
  specialization_id int fk

  created_at timestamp
}

appointments [icon:calendar, color: orange] {
  appointment_id serial pk
  patient_id int fk
  doctor_id int fk
  appointment_datetime timestamp
  status enum('BOOKED','CANCELLED','COMPLETED','NO_SHOW')

  created_at timestamp
}

consultations [icon:file-text, color: teal] {
  consultation_id serial pk
  appointment_id int fk nullable
  patient_id int fk
  doctor_id int fk
  visit_datetime timestamp
  diagnosis text
  notes text
  follow_up_date date

  created_at timestamp
}

diagnostic_tests [icon:clipboard, color: red] {
  test_id serial pk
  test_name varchar(150)
  category varchar(100)  // NEW (blood, radiology, etc.)
  description text
  cost decimal(10,2)
}

prescriptions [icon:list, color: red] {
  prescription_id serial pk
  consultation_id int fk
  test_id int fk
  instructions text
  status enum('PENDING','COMPLETED')

  created_at timestamp
}

reports [icon:file, color: cyan] {
  report_id serial pk
  prescription_id int fk
  patient_id int fk      // NEW (denormalized for fast queries)
  doctor_id int fk       // NEW (who handled case)
  generated_date timestamp
  result_summary text
  file_path text
}

payments [icon:credit-card, color: yellow] {
  payment_id serial pk
  patient_id int fk      // NEW (direct link for billing)
  consultation_id int fk nullable
  appointment_id int fk nullable
  amount decimal(10,2)
  payment_date timestamp
  payment_mode varchar(50)
  status enum('SUCCESS','FAILED','PENDING')

  created_at timestamp
}


// =========================
// RELATIONSHIPS
// =========================

// Patient & Doctor with Appointment
patients.patient_id < appointments.patient_id
doctors.doctor_id < appointments.doctor_id

// Patient & Doctor with Consultation
patients.patient_id < consultations.patient_id
doctors.doctor_id < consultations.doctor_id

// Appointment to Consultation
appointments.appointment_id < consultations.appointment_id

// Consultation to Test Prescription
consultations.consultation_id < prescriptions.consultation_id

// Test to Prescription
diagnostic_tests.test_id < prescriptions.test_id

// Prescription to Report
prescriptions.prescription_id < reports.prescription_id

// Reports direct links 
patients.patient_id < reports.patient_id
doctors.doctor_id < reports.doctor_id

// Payments
patients.patient_id < payments.patient_id
consultations.consultation_id < payments.consultation_id
appointments.appointment_id < payments.appointment_id

// Specialization
specializations.specialization_id < doctors.specialization_id
